#include <iostream>
using namespace std;

struct Student
{
   int NO; // 学号
   char *pName; // 姓名 
   double score; // 成绩
   bool bsex; // 性别
   double height; // 身高

   void printMsg(){ // 显示学生信息
      cout << "学号: " << NO << " 姓名: " << pName
           << " 性别: " << bsex << " 成绩: " << score 
           << " 身高: " << height << endl;
   }
};

int main()
{
   Student stu1;
   stu1.NO = 007;
   stu1.pName = "张三";
   stu1.score = 90.5;
   stu1.bsex = false;
   stu1.printMsg();
   
   Student stu2;
   stu2.NO = 008;
   stu2.pName = "李四";
   stu2.score = 80.5;
   stu2.bsex = true;
   stu2.printMsg();

   return 0;
}
