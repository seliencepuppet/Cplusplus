```c++
#include <iostream>
using namespace std;

class Student
{
public:
   int NO; // 学号
   char *pName; // 姓名 
   double score; // 成绩
   bool bsex; // 性别
   double height; // 身高
   
   void init(int nNo, char *pN, double ds, bool bs, double dh){ // 初始化成员变量
      NO = nNo;
      pName = pN;
      score = ds;
      bsex = bs;
      height = dh;
   }

   void printMsg(){ // 显示学生信息
      cout << "学号: " << NO << " 姓名: " << pName
           << " 性别: " << bsex << " 成绩: " << score 
           << " 身高: " << height << endl;
           
      if(score > 60.5){
         cout << "合格" << endl;
      }else{
         cout << "不合格" << endl;
      }
   }
};

int main()
{
   Student stu1;
   stu1.init(007, "张三", 90.5, false, 170.5);
   stu1.printMsg();
   
   Student stu2;
   stu2.init(008, "李四", 80.5, true, 180.3);
   stu2.printMsg();

   return 0;
}
```
